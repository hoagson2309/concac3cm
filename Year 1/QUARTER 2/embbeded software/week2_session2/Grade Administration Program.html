typedef struct {

    colour_t band2;
    colour_t multiplier;
} resistor_t;

const char *color_names[] = {
    "black", "brown", "red", "orange", "yellow",
    "green", "blue", "violet", "grey", "white",

};

// Case-insensitive string comparison
bool equal_strings_ci(const char *str1, const char *str2) {
    while (*str1 && *str2) {
        if (tolower(*str1) != tolower(*str2)) {
            return false;
        }
        str1++;
        str2++;
    }
    return *str1 == '\0' && *str2 == '\0';
}

// Check if a colour is valid for the first two bands
bool is_valid_digit_band(colour_t colour) {
    return colour >= BLACK && colour <= WHITE; // Valid colours for digit bands
}

// Returns the value of the resistor or NaN for invalid colours
double get_value(const resistor_t *resistor) {
    if (!is_valid_digit_band(resistor->band1) || !is_valid_digit_band(resistor->band2)) {
        return NAN; // Return "not a number" for invalid colours
    }

    // Compute the base value of the first two bands
    int base_value = (resistor->band1 * 10) + resistor->band2;

    if (resistor->multiplier == GOLD) {
        return base_value * 0.1; // Gold: 10^-1
    }
    if (resistor->multiplier == SILVER) {
        return base_value * 0.01; // Silver: 10^-2
    }
    if (resistor->multiplier >= BLACK && resistor->multiplier <= WHITE) {
        return base_value * pow(10, resistor->multiplier); // Standard: 10^n
    }

    return NAN; // Invalid multiplier
}

// Reads a colour from the user
colour_t read_colour(const char *prompt) {
    char input[20]; // Buffer for user input
    while (true) {
        printf("%s", prompt);
        scanf("%19s", input);

        for (int i = 0; i < sizeof(color_names) / sizeof(color_names[0]); i++) {
            if (equal_strings_ci(input, color_names[i])) {
                return (colour_t)i; // Return the matching colour
            }
        }

        printf("Invalid colour. Please try again.\n");
    }
}

int main(void) {
    resistor_t resistor;

    // Ask user to enter the colours of the resistor

    resistor.band2 = read_colour("Please enter the second colour: ");

    // Calculate the value of the resistor

    // Print the result
    if (isnan(value)) {
        printf("Invalid colours were specified\n");
    } else {
        printf("The value of the resistor is %.2lf ohms\n", value);
    }

}
